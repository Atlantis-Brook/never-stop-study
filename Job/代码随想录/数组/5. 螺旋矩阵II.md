class Solution {

	public int[][] generateMatrix(int n) {
	
		int[][] arr = new int[n][n];		
		arr[0][0] = 1;		
		int offset = 1, nextI = 0, nextJ = 0;		
		int dict = 0, count = 1;		
		int i, j;
	
		while (dict <= 2 * n - 1) {		
			for (j = nextJ; j < n - offset; j ++ ) {			
				arr[nextI][j] = count ++ ;			
			}			
			dict ++ ;			
			for (i = nextI; i < n - offset; i ++ ) {			
				arr[i][j] = count ++ ;			
			}			
			dict ++ ;
			for (; j > nextI; j -- ) {
				arr[i][j] = count ++ ;
			}
			dict ++ ;
			for (; i > nextJ; i -- ) {
				arr[i][j] = count ++ ;
			}
			dict ++ ;  
			offset ++ ;
			nextI ++ ;
			nextJ ++ ;
		}
		  
		if (n % 2 == 1) {
			arr[n / 2][n / 2] = n * n;
		}
		return arr;	  
	
	}

}
