# 209.长度最小的子数组
[LeetCode：27. 移除元素](https://leetcode.cn/problems/remove-element/)
- **双指针**
```java
class Solution {
	public int[] sortedSquares(int[] nums) {
	
		int n = nums.length;
		int[] ans = new int[nums.length];
		int head = 0, tail = n - 1, pos = n - 1;
		while (head <= tail) {
			int x = nums[head] * nums[head];	
			int y = nums[tail] * nums[tail];
			if (x > y) {
				ans[pos -- ] = x;
				head ++ ;
			} else {	
				ans[pos -- ] = y;	
				tail -- ;	
			}
		}	
		return ans;
	}
}
```
