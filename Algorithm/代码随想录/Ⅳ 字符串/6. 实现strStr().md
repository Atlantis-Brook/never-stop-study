# 28. 找出字符串中第一个匹配项的下标
[力扣题目链接](https://leetcode.cn/problems/find-the-index-of-the-first-occurrence-in-a-string/)
- **KMP算法**
```java
class Solution {
    public int strStr(String haystack, String needle) {
        int j = -1;
        int[] next = new int[needle.length()];
        getNext(next, needle);
        for (int i = 0; i < haystack.length(); i ++ ) {
            while (j >= 0 && haystack.charAt(i) != needle.charAt(j + 1)) {
                j = next[j];
            }
            if (haystack.charAt(i) == needle.charAt(j + 1)) {
                j ++ ;
            }
            if (j == needle.length() - 1) {
                return i - j;
            }
        }
        return -1;
    }

    public void getNext(int[] next, String template) {
        int j = -1;
        next[0] = j;
        for (int i = 1; i < template.length(); i ++ ) {
            while (j >= 0 && template.charAt(i) != template.charAt(j + 1)) {
                j = next[j];
            } 
            if (template.charAt(i) == template.charAt(j + 1)) {
                j ++ ;
            }
            next[i] = j;  
        }
    }
}
```
