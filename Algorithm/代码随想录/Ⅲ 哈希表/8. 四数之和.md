# 18. 四数之和
[力扣题目链接](https://leetcode.cn/problems/4sum/)
- **哈希法：**
```java
class Solution {
    public int fourSumCount(int[] nums1, int[] nums2, int[] nums3, int[] nums4) {
        int n = nums1.length;
        int res = 0;
        if (n == 0) {
            return 0;
        }
        Map<Integer, Integer> tmpLeftSum = new HashMap<>(); 
        Map<Integer, Integer> tmpRightSum = new HashMap<>(); 
        for (int i = 0; i < n; i ++ ) {
            for (int j = 0; j < n; j ++ ) {
                int sum = nums1[i] + nums2[j];
                int count = tmpLeftSum.getOrDefault(sum, 0);
                tmpLeftSum.put(sum, ++ count);
            }
        }
        for (int i = 0; i < n; i ++ ) {
            for (int j = 0; j < n; j ++ ) {
               int sum = 0 - nums3[i] - nums4[j];
                int count = tmpRightSum.getOrDefault(sum, 0);
                tmpRightSum.put(sum, ++ count);
            }
        }

        for (Map.Entry<Integer, Integer> entry : tmpRightSum.entrySet()) {
            int target = entry.getKey();
            int count = tmpLeftSum.getOrDefault(target, 0);
            res += count * entry.getValue();
        }
        return res;
    }
}
```
