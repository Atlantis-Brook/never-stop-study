# 707.设计链表
[力扣题目链接](https://leetcode.cn/problems/design-linked-list/)
- **单链表**
```java
class MyLinkedList {
    private int val;
    private MyLinkedList next;

    public MyLinkedList() {
        this.val = -1;
        this.next = null;
    }

    public void setVal(int val) {
        this.val = val;
    }
    
    public void setNext(MyLinkedList next) {
        this.next = next;
    }

    public int get(int index) {
        MyLinkedList p = this.next;
        while (p != null) {
            if (index == 0) {
                return p.val;
            }
            index -- ;
            p = p.next;
        }
        return -1;
    }
    
    public void addAtHead(int val) {
        MyLinkedList node = new MyLinkedList();
        node.setVal(val);
        node.next = this.next;
        this.next = node;
    }
    
    public void addAtTail(int val) {
        MyLinkedList node = new MyLinkedList();
        node.setVal(val);
        MyLinkedList p = this;
        while (p.next != null) {
            p = p.next;
        }
        p.next = node;
    }
    
    public void addAtIndex(int index, int val) {
        MyLinkedList node = new MyLinkedList();
        node.setVal(val);
        MyLinkedList p = this.next;
        MyLinkedList pre = this;
        while (p != null) {
            if (index == 0) {
                node.next = p;
                pre.next = node;
                return ;
            }
            index -- ;
            pre = p;
            p = p.next;
        }
        if (index == 0) {
            pre.next = node;
        }
        return;
    }
    
    public void deleteAtIndex(int index) {
        MyLinkedList p = this.next;
        MyLinkedList pre = this;
        while (p != null) {
            if (index == 0) {
                pre.next = p.next;
                p.next = null;
                return ;
            }
            index -- ;
            pre = p;
            p = p.next;
        }
        return ;
    }
}
  ```
